def changeLine(arr):
    changeX = 607
    changeY = 643
    for char in arr:
        unitFormReg.wait(,20)
        click(Location(changeX,changeY))
        changeX += 109
        textReg.wait(,20)
        if selectReg.exists(char):
            selectChar(char)

    match = selectReg.find(img)
    miniReg = Region(match.x,match.y,match.w,match.h+40)
    wait(1)
    miniReg.click()
