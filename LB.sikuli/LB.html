def changeRow(arr):
    changeX = 607
    changeY = 643
    for char in arr:
        unitFormReg.wait(,20)
        click(Location(changeX,changeY))
        changeX += 109
        textReg.wait(,20)
        if selectReg.exists(char):
            selectChar(char)

def selectChar(img):
    #Move the mouse away from the tooltips
    textReg.wait(,20)
    match = selectReg.find(img)
    hover(Location(match.x,185))
    miniReg = Region(match.x,match.y,match.w,match.h+40)
    wait(1)
    miniReg.click()
